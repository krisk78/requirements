96 template <typename T>
97 void Requirements<T>::add(const T& dependent, const T& requirement)
98 {
99 assert(!(dependent == requirement) && "A requirement can't be requested for object itself.");
100 // we must ensure the implicit requirement does not already exist
101 assert(!exists(dependent, requirement, true) && "(Implicit) requirement is already defined.");
102 if (!m_reflexive)
103 // opposite requirement is only allowed if reflexivity is activated, directly or indirectly
104 assert(!exists(requirement, dependent, true) && "Opposite requirement cannot be set while reflexivity is not allowed.");
105 m_requirements.insert({ dependent, requirement });
106 }

277 {
278 assert(has_requirements(dependent) && "No requirement exists for this argument.");
279 std::vector<std::vector<T>> result{};
280 auto reqs = requirements(dependent);
281 for (auto req : reqs)
282 {
283 std::vector<T> row{ dependent };
284 size_t count{ 0 };
285 if (has_requirements(req))
286 {
287 auto subreqs = all_requirements(req);
288 for (auto sub : subreqs)
289 {
290 if (sub[1] != dependent) // avoid infinite loop while reflexivity is active
291 {
292 for (auto itr : sub)
293 row.push_back(itr);
294 result.push_back(row);
295 row.clear();
296 row.push_back(dependent);
297 ++count;
298 }
299 }
300 }
301 if (count == 0)
302 {
303 row.push_back(req);
304 result.push_back(row);
305 }
306 }
307 return result;

463 bool Requirements<T>::_requires(const T& dependent, const T& requirement, const T* prev) const
464 {
465 bool result = exists(dependent, requirement);
466 if (!result && has_requirements(dependent))
467 {
468 auto reqs = requirements(dependent);
469 auto req = reqs.begin();
470 while (!result && req != reqs.end())

473 result = _requires(*req, requirement, &dependent);
